267 /**
268 * @brief Load an animated sprite and its animation from a TOML configuration file
269 * @details An animation sprite configuration file should look this:
270 * @code
271 * [data]
272 * spriteSheet="path to your spritesheet"
273 * steps=["idleRight", "idleLeft", "runRight", ...]
274 *
275 * [idleRight]
276 * framesCount=3
277 * frameTime=0.2
278 * [idleRight.frame0]
279 * left=0
280 * top=0
281 * width=15
282 * height=15
283 * # optionnal attribute indicating that the animation shouldn't
284 * # continue if it reach this frame, can only be specified once
285 * # per step
286 * end=true
287 * ...
288 *
289 * [idleLeft]
290 * copy="idleRight"
291 * invertX=true
292 *
293 * ...
294 * @endcode
295 * @details An animation loader is loading Animations from a TOML configuration file,
296 * but it doesn't hold an AnimatedSprite, which should be handled separately.
297 * You can retrieve animations from the loader by using `my_loader["animation_name"]`,
298 * and attach them to a pre-existing AnimatedSprite.
299 */

317
