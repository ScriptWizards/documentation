151 void AnimatedSprite::setFrame(std::size_t newFrame, bool resetTime)
152 {
153 if (m_animation)
154 {
155 // calculate new vertex positions and texture coordiantes
156 sf::IntRect rect = (*m_animation)[newFrame];
157 sf::FloatRect rectf = sf::FloatRect(rect);
158
159 m_vertices[0].position = sf::Vector2f(0.f, 0.f);
160 m_vertices[1].position = sf::Vector2f(0.f, rectf.height);
161 m_vertices[2].position = sf::Vector2f(rectf.width, rectf.height);
162 m_vertices[3].position = sf::Vector2f(rectf.width, 0.f);
163
164 float left = rectf.left + 0.0001f;
165 float right = left + rectf.width;
166 float top = rectf.top;
167 float bottom = top + rectf.height;
168
169 if (m_animation->isXInverted())
170 {
171 left = rectf.left + rectf.width;
172 right = rectf.left;
173 top = rectf.top;
174 bottom = top + rectf.height;
175 }
176
177 m_vertices[0].texCoords = sf::Vector2f(left, top);
178 m_vertices[1].texCoords = sf::Vector2f(left, bottom);
179 m_vertices[2].texCoords = sf::Vector2f(right, bottom);
180 m_vertices[3].texCoords = sf::Vector2f(right, top);
181 }
182
183 if (resetTime)
184 m_currentTime = sf::Time::Zero;
185 }

187 void AnimatedSprite::update(sf::Time deltaTime)
188 {
189 // if not paused and we have a valid animation and it's not the end frame
190 if (!m_isPaused && m_animation && !isEnd())
191 {
192 // add delta time
193 m_currentTime += deltaTime;
194
195 // if current time is bigger then the frame time advance one frame
196 if (m_currentTime >= m_animation->getFrameTime())
197 {
198 // reset time, but keep the remainder
199 m_currentTime = sf::microseconds(m_currentTime.asMicroseconds() % m_animation->getFrameTime().asMicroseconds());
200
201 // get next Frame index
202 if (m_currentFrame + 1 < m_animation->size())
203 m_currentFrame++;
204 else
205 {
206 // animation has ended
207 if (!m_isLooped)
208 m_isPaused = true;
209 else
210 m_currentFrame = 0; // reset to start
211 }
212
213 // set the current frame, not reseting the time
214 setFrame(m_currentFrame, false);
215 }
216 }

239 bool AnimationLoader::load(const std::string& configFile)
240 {
241 std::ifstream ifs(configFile);
242 toml::ParseResult pr = toml::parse(ifs);
243 ifs.close();
244
245 // check if the toml parser is in an error state or not
246 if (!pr.valid())
247 {
248 std::cerr << "AnimationLoader configuration error: " << pr.errorReason << std::endl;
249 return false;
250 }
251 else
252 m_config = pr.value;
253
254 // search for data section
255 if (m_config.find("data") == nullptr)
256 {
257 std::cerr << "AnimationLoader configuration needs a data section" << std::endl;
258 return false;
259 }
260
261 // find spritesheet
262 const toml::Value* spriteSheet = m_config.find("data.spriteSheet");
263 if (spriteSheet == nullptr || !spriteSheet->is<std::string>())
264 {
265 std::cerr << "AnimationLoader configuration: data.spriteSheet (string) missing" << std::endl;
266 return false;
267 }
268
269 // find steps list, used to find all the other steps of the animation
270 const toml::Value* steps = m_config.find("data.steps");
271 if (steps == nullptr || !steps->is<std::vector<std::string>>())
272 {
273 std::cerr << "AnimationLoader configuration: data.steps (array of strings) missing" << std::endl;
274 return false;
275 }
276
277 // register spritesheet
278 if (!m_spriteSheet.loadFromFile(spriteSheet->as<std::string>()))
279 {
280 std::cerr << "AnimationLoader configuration: couldn't load spriteSheet: " << spriteSheet->as<std::string>() << std::endl;
281 return false;
282 }
283
284 // iterate on the steps
285 std::vector<std::string> stepsData = steps->as<std::vector<std::string>>();
286 for (std::size_t i = 0, size = stepsData.size(); i < size; ++i)
287 {
288 // do we need to copy another animation?
289 const toml::Value* copy = m_config.find(stepsData[i] + ".copy");
290 if (copy == nullptr)
291 {
292 // number of frames in the animation
293 const toml::Value* framesCount = m_config.find(stepsData[i] + ".framesCount");
294 if (framesCount == nullptr || !framesCount->is<int>())
295 {
296 std::cerr << "AnimationLoader configuration: " << stepsData[i] << ".framesCount is missing" << std::endl;
297 return false;
298 }
299
300 // frame time
301 // search for frame duration (global, not per animation)
302 const toml::Value* frameTime = m_config.find(stepsData[i] + ".frameTime");
303 if (frameTime == nullptr || !frameTime->is<double>())
304 {
305 std::cerr << "AnimationLoader configuration: " << stepsData[i] << ".frameTime (double) missing" << std::endl;
306 return false;
307 }
308
309 m_animations[stepsData[i]] = Animation();
310 m_animations[stepsData[i]].setSpriteSheet(m_spriteSheet);
311 m_animations[stepsData[i]].setFrameTime(sf::seconds(static_cast<float>(frameTime->as<double>())));
312
313 // iterate on the frames
314 int framesCountData = framesCount->as<int>();
315 for (int frame = 0; frame < framesCountData; ++frame)
316 {
317 std::stringstream frameStr;
318 frameStr << "frame" << frame;
319
320 // find the positions for the sprite
321 int top, left, width, height;
322
323 toml::Value* check = m_config.find(stepsData[i] + "." + frameStr.str() + ".left");
324 if (check == nullptr || !check->is<int>())
325 {
326 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".left (integer) is missing" << std::endl;
327 return false;
328 }
329 else
330 left = check->as<int>();
331
332 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".top");
333 if (check == nullptr || !check->is<int>())
334 {
335 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".top (integer) is missing" << std::endl;
336 return false;
337 }
338 else
339 top = check->as<int>();
340
341 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".width");
342 if (check == nullptr || !check->is<int>())
343 {
344 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".width (integer) is missing" << std::endl;
345 return false;
346 }
347 else
348 width = check->as<int>();
349
350 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".height");
351 if (check == nullptr || !check->is<int>())
352 {
353 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".height (integer) is missing" << std::endl;
354 return false;
355 }
356 else
357 height = check->as<int>();
358
359 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".end");
360 bool end = false;
361 if (check != nullptr && check->is<bool>() && check->as<bool>())
362 end = true;
363
364 m_animations[stepsData[i]].addFrame(sf::IntRect(left, top, width, height));
365 if (end)
366 m_animations[stepsData[i]].setEnd(static_cast<int>(m_animations[stepsData[i]].size()) - 1);
367 }
368 }
369 else
370 {
371 if (!copy->is<std::string>())
372 {
373 std::cerr << "AnimationLoader configuration: " << stepsData[i] << ".copy should be a string" << std::endl;
374 return false;
375 }
376 m_animations[stepsData[i]] = m_animations[copy->as<std::string>()];
377
378 // do we need to invert the x axis of an animation?
379 const toml::Value* invertX = m_config.find(stepsData[i] + ".invertX");
380 if (invertX && invertX->is<bool>() && invertX->as<bool>())
381 m_animations[stepsData[i]].invertX();
382 }
383 }
384
385 return true;
386 }
