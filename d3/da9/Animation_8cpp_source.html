152 void AnimatedSprite::setFrame(std::size_t newFrame, bool resetTime)
153 {
154 if (m_animation)
155 {
156 // calculate new vertex positions and texture coordiantes
157 sf::IntRect rect = (*m_animation)[newFrame];
158 sf::FloatRect rectf = sf::FloatRect(rect);
159
160 m_vertices[0].position = sf::Vector2f(0.f, 0.f);
161 m_vertices[1].position = sf::Vector2f(0.f, rectf.height);
162 m_vertices[2].position = sf::Vector2f(rectf.width, rectf.height);
163 m_vertices[3].position = sf::Vector2f(rectf.width, 0.f);
164
165 float left = rectf.left + 0.0001f;
166 float right = left + rectf.width;
167 float top = rectf.top;
168 float bottom = top + rectf.height;
169
170 if (m_animation->isXInverted())
171 {
172 left = rectf.left + rectf.width;
173 right = rectf.left;
174 top = rectf.top;
175 bottom = top + rectf.height;
176 }
177
178 m_vertices[0].texCoords = sf::Vector2f(left, top);
179 m_vertices[1].texCoords = sf::Vector2f(left, bottom);
180 m_vertices[2].texCoords = sf::Vector2f(right, bottom);
181 m_vertices[3].texCoords = sf::Vector2f(right, top);
182 }
183
184 if (resetTime)
185 m_currentTime = sf::Time::Zero;
186 }

188 void AnimatedSprite::update(sf::Time deltaTime)
189 {
190 // if not paused and we have a valid animation and it's not the end frame
191 if (!m_isPaused && m_animation && !isEnd())
192 {
193 // add delta time
194 m_currentTime += deltaTime;
195
196 // if current time is bigger then the frame time advance one frame
197 if (m_currentTime >= m_animation->getFrameTime())
198 {
199 // reset time, but keep the remainder
200 m_currentTime = sf::microseconds(m_currentTime.asMicroseconds() % m_animation->getFrameTime().asMicroseconds());
201
202 // get next Frame index
203 if (m_currentFrame + 1 < m_animation->size())
204 m_currentFrame++;
205 else
206 {
207 // animation has ended
208 if (!m_isLooped)
209 m_isPaused = true;
210 else
211 m_currentFrame = 0; // reset to start
212 }
213
214 // set the current frame, not reseting the time
215 setFrame(m_currentFrame, false);
216 }
217 }

240 bool AnimationLoader::load(const std::string& configFile)
241 {
242 std::ifstream ifs(configFile);
243 toml::ParseResult pr = toml::parse(ifs);
244 ifs.close();
245
246 // check if the toml parser is in an error state or not
247 if (!pr.valid())
248 {
249 std::cerr << "AnimationLoader configuration error: " << pr.errorReason << std::endl;
250 return false;
251 }
252 else
253 m_config = pr.value;
254
255 // search for data section
256 if (m_config.find("data") == nullptr)
257 {
258 std::cerr << "AnimationLoader configuration needs a data section" << std::endl;
259 return false;
260 }
261
262 // find spritesheet
263 const toml::Value* spriteSheet = m_config.find("data.spriteSheet");
264 if (spriteSheet == nullptr || !spriteSheet->is<std::string>())
265 {
266 std::cerr << "AnimationLoader configuration: data.spriteSheet (string) missing" << std::endl;
267 return false;
268 }
269
270 // find steps list, used to find all the other steps of the animation
271 const toml::Value* steps = m_config.find("data.steps");
272 if (steps == nullptr || !steps->is<std::vector<std::string>>())
273 {
274 std::cerr << "AnimationLoader configuration: data.steps (array of strings) missing" << std::endl;
275 return false;
276 }
277
278 // register spritesheet
279 sf::Texture tex;
280 if (!tex.loadFromFile(spriteSheet->as<std::string>()))
281 {
282 std::cerr << "AnimationLoader configuration: couldn't load spriteSheet: " << spriteSheet->as<std::string>() << std::endl;
283 return false;
284 }
285 m_textureID = spriteSheet->as<std::string>();
286 GameObjects::TextureManager::get().add(m_textureID, std::move(tex));
287
288 // iterate on the steps
289 std::vector<std::string> stepsData = steps->as<std::vector<std::string>>();
290 for (std::size_t i = 0, size = stepsData.size(); i < size; ++i)
291 {
292 // do we need to copy another animation?
293 const toml::Value* copy = m_config.find(stepsData[i] + ".copy");
294 if (copy == nullptr)
295 {
296 // number of frames in the animation
297 const toml::Value* framesCount = m_config.find(stepsData[i] + ".framesCount");
298 if (framesCount == nullptr || !framesCount->is<int>())
299 {
300 std::cerr << "AnimationLoader configuration: " << stepsData[i] << ".framesCount is missing" << std::endl;
301 return false;
302 }
303
304 // frame time
305 // search for frame duration (global, not per animation)
306 const toml::Value* frameTime = m_config.find(stepsData[i] + ".frameTime");
307 if (frameTime == nullptr || !frameTime->is<double>())
308 {
309 std::cerr << "AnimationLoader configuration: " << stepsData[i] << ".frameTime (double) missing" << std::endl;
310 return false;
311 }
312
313 m_animations[stepsData[i]] = Animation();
314 m_animations[stepsData[i]].setSpriteSheet(GameObjects::TextureManager::get()[m_textureID]);
315 m_animations[stepsData[i]].setFrameTime(sf::seconds(static_cast<float>(frameTime->as<double>())));
316
317 // iterate on the frames
318 int framesCountData = framesCount->as<int>();
319 for (int frame = 0; frame < framesCountData; ++frame)
320 {
321 std::stringstream frameStr;
322 frameStr << "frame" << frame;
323
324 // find the positions for the sprite
325 int top, left, width, height;
326
327 toml::Value* check = m_config.find(stepsData[i] + "." + frameStr.str() + ".left");
328 if (check == nullptr || !check->is<int>())
329 {
330 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".left (integer) is missing" << std::endl;
331 return false;
332 }
333 else
334 left = check->as<int>();
335
336 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".top");
337 if (check == nullptr || !check->is<int>())
338 {
339 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".top (integer) is missing" << std::endl;
340 return false;
341 }
342 else
343 top = check->as<int>();
344
345 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".width");
346 if (check == nullptr || !check->is<int>())
347 {
348 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".width (integer) is missing" << std::endl;
349 return false;
350 }
351 else
352 width = check->as<int>();
353
354 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".height");
355 if (check == nullptr || !check->is<int>())
356 {
357 std::cerr << "AnimationLoader configuration: " << stepsData[i] << "." << frameStr.str() << ".height (integer) is missing" << std::endl;
358 return false;
359 }
360 else
361 height = check->as<int>();
362
363 check = m_config.find(stepsData[i] + "." + frameStr.str() + ".end");
364 bool end = false;
365 if (check != nullptr && check->is<bool>() && check->as<bool>())
366 end = true;
367
368 m_animations[stepsData[i]].addFrame(sf::IntRect(left, top, width, height));
369 if (end)
370 m_animations[stepsData[i]].setEnd(static_cast<int>(m_animations[stepsData[i]].size()) - 1);
371 }
372 }
373 else
374 {
375 if (!copy->is<std::string>())
376 {
377 std::cerr << "AnimationLoader configuration: " << stepsData[i] << ".copy should be a string" << std::endl;
378 return false;
379 }
380 m_animations[stepsData[i]] = m_animations[copy->as<std::string>()];
381
382 // do we need to invert the x axis of an animation?
383 const toml::Value* invertX = m_config.find(stepsData[i] + ".invertX");
384 if (invertX && invertX->is<bool>() && invertX->as<bool>())
385 m_animations[stepsData[i]].invertX();
386 }
387 }
388
389 return true;
390 }
