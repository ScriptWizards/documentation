59 void Frame()
60 {
61 auto& prevEntry = _entries[_currentEntry];
62 _currentEntry = (_currentEntry + 1) % _bufferSize;
63 prevEntry._frameEnd = _entries[_currentEntry]._frameStart = std::chrono::system_clock::now();
64 }
65
66 void Begin(Stage stage)
67 {
68 assert(_currentLevel < 255);
69 auto& entry = _entries[_currentEntry]._stages[stage];
70 entry._level = _currentLevel;
71 _currentLevel++;
72 entry._start = std::chrono::system_clock::now();
73 entry._finalized = false;
74 }
75
76 void End(Stage stage)
77 {
78 assert(_currentLevel > 0);
79 auto& entry = _entries[_currentEntry]._stages[stage];
80 assert(!entry._finalized);
81 _currentLevel--;
82 assert(entry._level == _currentLevel);
83 entry._end = std::chrono::system_clock::now();
84 entry._finalized = true;
85 }
86
87 unsigned char GetCurrentEntryIndex() const
88 {
89 return (_currentEntry + _bufferSize - 1) % _bufferSize;
90 }
91
92 static const unsigned char _bufferSize = 100;
93 std::array<Entry, _bufferSize> _entries;
94
95 private:
96 unsigned char _currentEntry = _bufferSize - 1;
97 unsigned char _currentLevel = 0;
98 };

117 static void ProfilerValueGetter(float* startTimestamp, float* endTimestamp, unsigned char* level, const char** caption, const void* data, int idx)
118 {
119 auto entry = reinterpret_cast<const Profiler::Entry*>(data);
120 auto& stage = entry->_stages[idx];
121
122 if (startTimestamp)
123 {
124 std::chrono::duration<float, std::milli> fltStart = stage._start - entry->_frameStart;
125 *startTimestamp = fltStart.count();
126 }
127
128 if (endTimestamp)
129 {
130 *endTimestamp = stage._end.time_since_epoch().count() / 1000000.0f;
131
132 std::chrono::duration<float, std::milli> fltEnd = stage._end - entry->_frameStart;
133 *endTimestamp = fltEnd.count();
134 }
135
136 if (level)
137 *level = stage._level;
138
139 if (caption)
140 *caption = stageNames[idx];
141 }
