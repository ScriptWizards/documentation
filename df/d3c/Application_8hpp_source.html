73 /**
74 * @brief The main component of the library, handling the scenes and running the app
75 * @details Debug mode is showing an imgui debug interface, with a profiler to know
76 what took time to execute. When scripting is turned on, it's expecting
77 to find a main script named "main.ark" defining those functions:
78 * @code
79 * (let onLoad (fun () {
80 * # code executed during loading
81 * }))
82 * (let onQuit (fun () {
83 * # code executed when closing
84 * }))
85 * @endcode

173 /**
174 * @brief Add a scene to the application
175 * @details Create the scene in place.
176 *
177 * @tparam S The type of the scene
178 * @tparam Args
179 * @param args Arguments for the constructor of the scene
180 * @return Application&
181 */
182 template <typename S, typename... Args>
183 Application& add(Args&&... args)
184 {
185 m_sceneManager.add<S>(std::forward<Args>(args)...);
186 return *this;
187 }

195 void run();
