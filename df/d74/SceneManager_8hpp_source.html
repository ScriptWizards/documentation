73 /**
74 * @brief Set the Current scene
75 *
76 * @tparam S The type of the scene
77 * @param data Default to nullptr. Points to data owned by the scene calling setCurrent, to give to the next scene
78 * @return SceneManager&
79 */
80 template <typename Scene>
81 SceneManager& setCurrent(void* data = nullptr)
82 {
83 const std::type_info& type(typeid(Scene));
84 int id = m_hashCodeToSceneId[type.hash_code()];
85
86 if (0 <= id && id < static_cast<int>(m_scenes.size()))
87 {
88 if (m_current != -1)
89 m_scenes[m_current]->setState(State::Stopped);
90 m_current = id;
91 m_scenes[m_current]->setState(State::Running);
92
93 // call the onChange method of the scene to notify it
94 // that the current running scene has changed
95 m_scenes[m_current]->onChange(data);
96 }
97 else
98 spdlog::warn("<SceneManager> Could not set current scene to {}: scene out of bounds [0, {}[", id, m_scenes.size());
99 return *this;
100 }

138 */
139 template <typename S, typename... Args>
140 void add(Args&&... args)
141 {
142 m_scenes.push_back(std::make_unique<S>(static_cast<int>(m_scenes.size()), std::forward<Args>(args)...));
143 // register the scene manger in the newly created scene
144 m_scenes.back().get()->m_sceneManager = this;
145
146 const std::type_info& type(typeid(S));
147 m_hashCodeToSceneId[type.hash_code()] = static_cast<int>(m_scenes.size()) - 1;
148 }
