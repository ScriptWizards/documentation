71 /**
72 * @brief Set the Current scene
73 *
74 * @tparam S The type of the scene
75 * @param data Default to nullptr. Points to data owned by the scene calling setCurrent, to give to the next scene
76 * @return SceneManager&
77 */
78 template <typename Scene>
79 SceneManager& setCurrent(void* data = nullptr)
80 {
81 const std::type_info& type(typeid(Scene));
82 int id = m_hashCodeToSceneId[type.hash_code()];
83
84 if (0 <= id && id < static_cast<int>(m_scenes.size()))
85 {
86 if (m_current != -1)
87 m_scenes[m_current]->setState(State::Stopped);
88 m_current = id;
89 m_scenes[m_current]->setState(State::Running);
90
91 // call the onChange method of the scene to notify it
92 // that the current running scene has changed
93 m_scenes[m_current]->onChange(data);
94 }
95 return *this;
96 }

122 /**
123 * @brief Register a scene and return its identifier
124 *
125 * @tparam S The type of the scene
126 * @tparam Args
127 * @param args Arguments for the constructor of the scene
128 */
129 template <typename S, typename... Args>
130 void add(Args&&... args)
131 {
132 m_scenes.push_back(std::make_unique<S>(static_cast<int>(m_scenes.size()), std::forward<Args>(args)...));
133 // register the scene manger in the newly created scene
134 m_scenes.back().get()->m_sceneManager = this;
135
136 const std::type_info& type(typeid(S));
137 m_hashCodeToSceneId[type.hash_code()] = static_cast<int>(m_scenes.size()) - 1;
138 }
139
140 private:
141 std::vector<std::unique_ptr<Scene>> m_scenes;
