205  /**
206  * @brief Utility to create widget out of the "layout" box
207  * @details Useful when it comes to try something out without having to drop
208  * a layout and some heavy configuration code. Example:
209  * @code
210  * auto button = sgl::Widgets::make<sgl::Widgets::Button>(sf::IntRect(12, 12, 48, 16));
211  * @endcode
212  *
213  * @tparam W The widget typename
214  * @tparam Args
215  * @param args Arguments for the widget, appart from the automatic ones (id and parent)
216  * @return W
217  */
218  template <typename W, typename... Args>
219  W make(Args&&... args)
220  {
221  return W(/* id */ 0, /* parent */ nullptr, std::forward<Args>(args)...);
222  }
